Usage Examples

Basic Usage

// Initialize the cache library const cache = new nooblyjsCaching({ instanceName: 'default', debug: true }); // Store data cache.put('user:123', { name: 'John', age: 30 }) .then(() => console.log('Stored!')) .catch(err => console.error('Failed:', err)); // Retrieve data cache.get('user:123') .then(data => console.log('Retrieved:', data)) .catch(err => console.error('Failed:', err)); // Delete data cache.delete('user:123') .then(() => console.log('Deleted!')) .catch(err => console.error('Failed:', err)); // Check if key exists cache.exists('user:123') .then(exists => console.log(exists ? 'Exists' : 'Not found'));

Batch Operations

// Store multiple values cache.putBatch({ 'user:123': { name: 'John' }, 'user:456': { name: 'Jane' }, 'config:app': { theme: 'dark' } }).then(() => console.log('All stored!')); // Retrieve multiple values cache.getBatch(['user:123', 'user:456', 'config:app']) .then(data => console.log('Retrieved:', data)); // Delete multiple values cache.clear(['user:123', 'user:456', 'config:app']) .then(() => console.log('All cleared!'));

Service Information

// Check cache service status cache.status() .then(status => console.log('Status:', status)); // List available instances cache.listInstances() .then(instances => console.log('Instances:', instances));
